def day_of_week(year, month, day):
  m = mod(month-3, 4800)
  y = mod(year +m/12, 400)
  m %= 12
  return (y + y/4 - y/100 + (13*m +2)/5 + day + 2) % 7

def mod(x, y):
  x %= y
  if (y > 0 and x < 0) or (y < 0 and x > 0):
    x += y
  return x
